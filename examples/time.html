Pikaday - Time Extension examples

A refreshing JavaScript Datepicker — lightweight, no dependencies, modular CSS.

var picker = new Pikaday({
    field               : document.getElementById('datepicker-default')
});
var picker = new Pikaday({
    field               : document.getElementById('datepicker-dateonly'),
    showTime            : false
});
var picker = new Pikaday({
    field               : document.getElementById('datepicker-24sec'),
    showSeconds         : true,
    use24hour           : true
});
var picker = new Pikaday({
    field               : document.getElementById('datepicker-defaultTime-3pm'),
    defaultTime         : '15:00:00'
});
var picker = new Pikaday({
    field               : document.getElementById('datepicker-defaultTime-null'),
    defaultTime         : null,
    showSeconds         : true
});
var picker = new Pikaday({
    field               : document.getElementById('datepicker-steps'),
    showSeconds         : true,
    defaultTime         : null,
    hourStep            : 2,
    minuteStep          : 5,
    secondStep          : 15,
    i18n                : {
      minutePlaceholder   : 'Minute'          // the default is "Min"
    }
});

What is this?

Since version 1.0 Pikaday is a stable and battle tested date-picker. Feel free to use it however you like but please report any bugs or feature requests to the GitHub issue tracker, thanks!
